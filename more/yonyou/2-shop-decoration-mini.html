# 一、项目介绍

项目来源

  • 来源于公司项目
  • 简易版店铺装修
# 介绍背景

我司 商业创新平台（BIP）

所在部门 营销云

提供的 B2C 服务中，针对 B2C 商城 -> 移动店铺装修

有幸接触到这一项目，想弄清楚这一庞大项目中的原理，所以做了一个简易版的店铺装修，我称之为 mini 版店铺装修，

项目介绍

所谓店铺装修，就是用户可以在 PC 端进行移动页面的制作，只需要通过简单的拖拽就可以实现页面的编辑，属于用户高度自定义的功能。最终编辑的结果，可以在 H5、小程序进行展示推广。 有赞美业店铺装修前端解决方案 (opens new window)

# 二、项目核心

# 2.1 设计态与运行态

  1. 说明：管理端（设计态）、移动端（运行态）

  2. 设计态与运行态的介绍：设计态通过属性可以控制展示的内容和展示形式（样式），点击保存之后可以将设置的状态保存起来（存到数据库中），进入到移动端页面（也就是运行态），通过从服务端获取配置信息，渲染页面。

  3. 设计态：设计态（管理端）用户可以在 PC 端对移动页面进行制作。

    • 从布局上分为 TopMenu LeftMenu MainArea Property
    • designer 文件夹内 Designer
    • LeftMenu MainArea 可以拖拽

import React, { useEffect } from "react";
import { connect } from "react-redux";
import { bindActionCreators } from "redux";
import { parseContainer } from "../../../helpers/render";
import * as layoutActions from "../../../redux/modules/pageConfig/layout";
import { setLayoutAndConfig } from "../../../redux/modules/pageConfig";

function MainArea(props) {
  useEffect(() => {
    props.setLayoutAndConfig();
  }, []); // eslint-disable-line
  const { layout, config } = props.pageConfig;
  const component = parseContainer(layout, config);
  return component;
}

function mapStateToProps(state) {
  return {
    pageConfig: state.pageConfig,
  };
}

function mapDispatchToProps(dispatch) {
  return {
    layoutActions: bindActionCreators(layoutActions, dispatch),
    setLayoutAndConfig: bindActionCreators(setLayoutAndConfig, dispatch),
  };
}

export default connect(mapStateToProps, mapDispatchToProps)(MainArea);

# 2.2 数据驱动与渲染

# （1）功能

  1. 后端预置数据

  2. 管理端可以进行页面装修

    1. 装修页面，请求接口 /mock/leftMenu /mock/pageData，拿到数据后，渲染 LeftMenu MainArea Property
    2. 点击中间区域某组件，能在右侧显示对应的属性信息 active
    3. 在右侧进行属性设置，中间区域会实时根据设置的属性进行渲染
      • MainArea 与组件是父子关系，父组件通过属性给子组件传了一个方法 updateConfig，当子组件 this.props.updateConfig
    4. 删除指定组件
    5. 重置，恢复到原来状态
    6. 保存，将当前设置的组件和属性保存到数据库中

  1. 移动端渲染

    • 调取 /mock/pageData 接口获取数据，根据配置信息渲染页面

  2. 其他

    • 中间件
      • redux-thunk
      • logger

const initialState = {
  activeType: "",

const types = {
  SET_ACTIVE: "SET_ACTIVE",

const active = (state = initialState, action) => {
  switch (action.type) {
    case types.SET_ACTIVE:
      return action.data;
    default:
      return state;
  }

export default active;

export const actions = {
  setActive: (type, key) => ({
    type: types.SET_ACTIVE,
    data: { activeType: type, activeKey: key },
  }),
};

  return function(dispatch) {
    dispatch(actions.setActive(type, key));
  };
};

import * as Components from "../common/components";

const ComponentsMap = {};
for (let attr in Components) {
  ComponentsMap[attr.toLocaleLowerCase()] = Components[attr];
}

export const parseContainer = (layout, config, callback) => {
  return (
    <div>
      {layout.map((item) => parseControl(item, config[item.key], callback))}
    </div>
  );
};

const parseControl = (component, config, callback) => {
  const { key, type } = component;
  const ComName = ComponentsMap[type];
  if (!ComName) return <h4 key={key}>Component {type} doesn't realize yet</h4>;
  return (
    <div key={key}>
      <ComName
        config={config}
        designer={callback ? true : false}
        onClick={callback ? () => callback(type, key) : undefined}
      />
    </div>
  );
};

import * as Properties from "../../properties";

const PropertiesMap = {};

  PropertiesMap[attr.toLocaleLowerCase()] = Properties[attr];

const parseProperty = (type, key, config) => {
  const { updateConfig } = props.configActions;
  if (!type) return <div>您没有点击任何组件~</div>;

      updateConfig={(newConfig) => updateConfig(type, key, newConfig)}
    />
  );
};

# 2.3 拖拽

# （1）插件

react-beautiful-dnd (opens new window)

  • 25.7k
  • 相比于 react-dnd，react-beautiful-dnd 更适用于列表之间拖拽的场景，支持移动端，且较为容易上手。

react-draggable

  • 7.2k
  • react-draggable 是 react 中的一个可拖动组件库。他可以指定 x , y 轴的拖动，可以监听一系列拖动事件

拖拽依赖第三方库 react-beautiful-dnd，可以实现组件动态排序的效果。

# （2）应用

react-beautiful-dnd (opens new window) 详细的介绍和使用可以参考官网及相关技术文章，这里只介绍项目中用到的。

项目中需要用到拖拽的有 LeftMenu 和 MainArea

Designer

  • DragDropContext 包裹可以拖拽的范围
  • 支持的事件：
    • onDragStart：拖拽开始的回调
    • onDragUpdate：拖拽中的回调
    • onDragEnd：拖拽结束时的回调
import { DragDropContext } from "react-beautiful-dnd";
import { reorderTest, moveTest } from "../../../helpers/decorationDnd";

function Designer(props) {
  // 拖拽结束触发事件
  const onDragEnd = (result) => {
    const { leftMenu, layout } = props;
    const {
      leftMenuActions: { setLeftMenu },
      layoutActions: { setLayout },
    } = props;
    const { source, destination } = result;

    // 在droppable外释放
    if (!destination) {
      return;
    }

    // 同列表拖拽排序
    if (source.droppableId === destination.droppableId) {
      if (source.droppableId === "leftMenu") {
        const leftmenuNew = reorderTest(
          leftMenu,
          source.index,
          destination.index
        );
        setLeftMenu(leftmenuNew);
      }
      if (source.droppableId === "layout") {
        const layoutNew = reorderTest(layout, source.index, destination.index);
        setLayout(layoutNew);
      }
    } else {
      // 不同列表拖拽
      if (source.droppableId === "layout") {
        return;
      }
      const result = moveTest(
        props[source.droppableId],
        props[destination.droppableId],
        source,
        destination
      );
      setLayout(result);
    }
  };
  return (
    <>
      <TopMenu />
      <div className="designer">
        <DragDropContext onDragEnd={onDragEnd}>
          <LeftMenu />
          <MainArea />
        </DragDropContext>
        <Property />
      </div>
    </>
  );
}

import uuid from "uuid";

// 拖拽模块样式
export const getItemStyleTest = (isDragging, draggableStyle) => ({
  margin: "0 0 8px 0",
  width: "100px",
  padding: "10px",
  textAlign: "center",
  // 拖拽时改变背景色
  background: isDragging ? "#79c4ec" : "#fff",

  // 拖拽时列表的元素会根据拖拽的位置移动
  ...draggableStyle,
});

// 拖拽列表样式
export const getListStyleTest = (isDraggingOver) => ({
  // background: isDraggingOver ? "lightblue" : "lightgrey",
  width: "100px",
});

// 同列表重新排序
export const reorderTest = (list, startIndex, endIndex) => {
  const result = Array.from(list);
  const [removed] = result.splice(startIndex, 1);
  result.splice(endIndex, 0, removed);
  return result;
};

// 处理列表间的数据变化
export const moveTest = (
  source,
  destination,
  droppableSource,
  droppableDestination
) => {
  const destClone = Array.from(destination);
  // 获取 源列表 拖拽的数据
  const choosed = source[droppableSource.index];
  // 将拖拽的数据根据index添加到放置列表
  destClone.splice(droppableDestination.index, 0, {
    type: choosed.type,
    key: choosed.type + uuid(),
  });
  return destClone;
};

需要在 LeftMenu、MainArea 中都引入 Droppable、Draggable，

  • Droppable (opens new window)：构建一个可以被拖拽放入的区域块。

  • Droppable：**可被拖拽的元素，**Draggable 必须始终包含在 Droppable 中，可以在 Droppable 内重新排序 Draggable 或移动到另一个 Droppable 。

  • React 实用库：react-beautiful-dnd (opens new window)

designer/LeftMenu

import { Droppable, Draggable } from "react-beautiful-dnd";
import {
  getItemStyleTest,
  getListStyleTest,
} from "../../../helpers/decorationDnd";

function LeftMenu(props) {
  useEffect(() => {
    props.leftMenuActions.leftMenuInit();
  }, []); // eslint-disable-line
  const map = {
    header: <CreditCardOutlined />,
    search: <SearchOutlined />,
    footer: <DatabaseOutlined />,
    noticebartest: <SoundOutlined />,
    swiperdemo: <InsertRowBelowOutlined />,
    imagedemo: <FileImageOutlined />,
    userlist: <UserOutlined />,
  };
  const height = document.body.clientHeight - 50;
  return (
    <div className="left_menu" style={{ height: height + "px" }}>
      <h3>基础组件</h3>
      <Droppable droppableId="leftMenu">
        {(provided, snapshot) => (
          <div
            ref={provided.innerRef}
            style={getListStyleTest(snapshot.isDraggingOver)}
          >
            {props.leftMenu.map((item, index) => (
              <Draggable
                key={item.index}
                draggableId={item.index}
                index={index}
              >
                {(provided, snapshot) => (
                  <div
                    ref={provided.innerRef}
                    {...provided.draggableProps}
                    {...provided.dragHandleProps}
                    style={getItemStyleTest(
                      snapshot.isDragging,
                      provided.draggableProps.style
                    )}
                  >
                    <div style={{ fontSize: "35px" }}>{map[item.type]}</div>
                    <div>{item.name}</div>
                  </div>
                )}
              </Draggable>
            ))}
            {provided.placeholder}
          </div>
        )}
      </Droppable>
    </div>
  );
}

import { Droppable, Draggable } from "react-beautiful-dnd";
import { parseContainer } from "../../../helpers/render";

function MainArea(props) {
  useEffect(() => {
    props.setLayoutAndConfig();
  }, []); // eslint-disable-line
  const onClick = (type, key) => {
    props.activeActions.setActive(type, key);
  };
  const {
    layout,
    config,
    active: { activeKey },
  } = props.pageConfig;
  const component = parseContainer(layout, config, onClick);
  const height = document.body.clientHeight;
  return (
    <div className="main_area_dnd" style={{ height: height - 80 + "px" }}>
      <div className="main_content" style={{ height: height - 120 + "px" }}>
        <Droppable droppableId="layout">
          {(provided, snapshot) => (
            <div
              ref={provided.innerRef}
              // style={{ width: "300px", position: "relative" }}
              style={{ position: "relative" }}
            >
              {component.props.children.map((item, index) => (
                <Draggable key={item.key} draggableId={item.key} index={index}>
                  {(provided, snapshot) => (
                    <div
                      ref={provided.innerRef}
                      {...provided.draggableProps}
                      {...provided.dragHandleProps}
                    >
                      {/* {item.key === activeKey && renderTool(activeKey)} */}
                      {item}
                    </div>
                  )}
                </Draggable>
              ))}
              {provided.placeholder}
            </div>
          )}
        </Droppable>
      </div>
    </div>
  );
}
