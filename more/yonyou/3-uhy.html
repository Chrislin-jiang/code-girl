# 会员中心

# 一、项目介绍

会员中心：

  • 会员中心的主要特性全渠道（电商平台、自建商城、微信、线下门店等）会员的集成管理。
  • 主要可以将其分为 移动端、管理端。

本项目是对会员中心移动端进行重构，主要包括以下内容：

  • 更新 UI 图、UE 图，以适应时代的发展，符合用户的审美；
  • 根据需求增加一些新特性或功能；
  • 前端搭建一套适用于会员中心移动端应用场景和需求的架构。

我在整个项目中负责的内容是：会员中心首页开发，包括设计态与运行态。

# 2.2 设计态公共组件

# (1) UploadImg

封装基础（公共）组件，首先分析这个基础组件接受哪些参数

参数 类型 说明 是否必传
imgUrl string 图片路由 是
comKey string 二级属性 是
id number - 否
uploadTip ReactNode 提示 是
onChange function 改变图片路由的方法 是
import React, { useEffect } from "react";
import { message } from "antd";
import UploadCommon from "../../../others/UploadCommon";
import { checkURL } from "../../../helpers/util";
import { getFileServerUrl } from "../../../helpers/fetch";

const imgServerUrl = "xxx";
const uploadServerUrl = "xxx";
let fileServerUrl = "xxx";

const beforeUpload = (file) => {
  const imgtypes = [
    "image/jpg",
    "image/jpeg",
    "image/png",
    "image/pjpeg",
    "image/gif",
    "image/bmp",
    "image/x-png",
  ];
  if (imgtypes.indexOf(file.type) == -1) {
    message.warning("仅支持png、jpg、gif格式，请重新选择！");
    return false;
  }
  const isLt2M = file.size / (1024 * 1024) > 2;
  if (isLt2M) {
    message.warning("上传图片超过限制，最大不超过2M");
    return false;
  }
  return true;
};

const getImageUrl = (imageUrl) => {
  if (fileServerUrl != "") {
    return fileServerUrl + imageUrl;
  }
  return imgServerUrl + imageUrl;
};

// 上传图片
const UploadImg = (props) => {
  const { comKey, id, imgUrl, uploadTip } = props || {};
  const { onChange } = props || {};
  useEffect(() => {
    // 获取文件服务器地址
    getFileServerUrl().then((res) => {
      if (res) {
        fileServerUrl = res.data;
      } else {
        message.error("获取文件服务器地址失败");
      }
    });
  }, []);
  const onUploadDone = (args) => {
    const file = args.file;
    let newImgUrl = checkURL(file.response.data)
      ? file.response.data
      : getImageUrl(file.response.data);
    if (comKey === "payingMember_imgUrl") {
      onChange(newImgUrl);
    } else {
      onChange(comKey, id, "icon", newImgUrl);
    }
  };
  const onDelete = () => {
    if (comKey === "payingMember_imgUrl") {
      onChange(null);
    } else {
      onChange(comKey, id, "icon", null);
    }
  };
  return (
    <div className="col-flex">
      <UploadCommon
        action={uploadServerUrl}
        showUploadList={false}
        onUploadDone={(args) => onUploadDone(args)}
        imgUrl={imgUrl}
        onDelete={onDelete}
        beforeUpload={(file) => beforeUpload(file)}
        showDelete={true}
      />
      <div className="col-tips">{uploadTip}</div>
    </div>
  );
};

export default UploadImg;

# (2) BtnGroup

参数

参数 类型 说明 是否必传
index number 自定义 xxx 索引 是
length number 自定义 xxx 长度，（在 length > 1 时，才显示 - 按钮） 是

delCustomData function 删除当前的自定义功能 是

定义

import React from "react";

// 自定义功能 按钮 + -
const BtnGroup = (props) => {
  const { index, length, addData } = props;
  const { addCustomData, delCustomData } = props;
  return (
    <div className="advert-temp-btn">
      <span
        className="btn-action-group"
        onClick={() => addCustomData(index, addData)}
      >
        <i className="uhyfont icon-self-jia"></i>
      </span>
      {length > 1 && (
        <span className="btn-action-group" onClick={() => delCustomData(index)}>
          <i className="uhyfont icon-self-jian"></i>
        </span>
      )}
    </div>
  );
};
export default BtnGroup;

定义

import React from "react";
import { Row, Checkbox, Input, Tooltip, message } from "antd";
import UploadImg from "./UploadImg";
import { checkURL } from "src/common/helpers/util";

// 活动任务（activityAndTask 固定的活动、weChatTask 通过接口请求到的微信活动） 互动娱乐
const PropertyList = (props) => {
  const { data, activityMap, comKey, name, icon, description, btn, targetUrl } =
    props || {};
  const { onChange } = props;
  return (
    <div>
      {activityMap && (
        <Row>
          <Checkbox
            checked={data.isShow}
            onChange={(e) =>
              onChange(comKey, data.id, "isShow", e.target.checked)
            }
          >
            {activityMap[data.type]}
          </Checkbox>
        </Row>
      )}
      {data.isShow && (
        <div className="two-space">
          {name?.isNeed && (
            <Row>
              <Checkbox
                checked={data.name_isShow}
                onChange={(e) =>
                  onChange(comKey, data.id, "name_isShow", e.target.checked)
                }
              >
                <label className="label-control">标题名称</label>
                <Tooltip
                  overlayClassName="tooltip-character-limit"
                  placement="top"
                  title={<span>{`最多${name?.maxLength}个字符`}</span>}
                  trigger="focus"
                >
                  <Input
                    onChange={(e) =>
                      onChange(comKey, data.id, "name", e.target.value)
                    }
                    value={data.name}
                    disabled={!data.name_isShow}
                    maxLength={name?.maxLength}
                  />
                </Tooltip>
              </Checkbox>
            </Row>
          )}
          {icon?.isNeed && (
            <Row>
              <Checkbox
                checked={data.icon_isShow}
                onChange={(e) =>
                  onChange(comKey, data.id, "icon_isShow", e.target.checked)
                }
              >
                <label className="label-control">图标</label>
              </Checkbox>
              {/* <UploadImg /> */}
              {data.icon_isShow && (
                <UploadImg
                  imgUrl={data.icon}
                  comKey={comKey}
                  id={data.id}
                  uploadTip={
                    <p>建议尺寸宽度228px，仅支持PNG、JPG、JPEG，最大不超过2M</p>
                  }
                  onChange={onChange}
                />
              )}
            </Row>
          )}
          {description?.isNeed && (
            <Row>
              <Checkbox
                checked={data.description_isShow}
                onChange={(e) =>
                  onChange(
                    comKey,
                    data.id,
                    "description_isShow",
                    e.target.checked
                  )
                }
              >
                <label className="label-control">辅助说明</label>
                <Tooltip
                  overlayClassName="tooltip-character-limit"
                  placement="top"
                  title={<span>{`最多${description?.maxLength}个字符`}</span>}
                  trigger="focus"
                >
                  <Input
                    onChange={(e) =>
                      onChange(comKey, data.id, "description", e.target.value)
                    }
                    value={data.description}
                    disabled={!data.description_isShow}
                    maxLength={description?.maxLength}
                  />
                </Tooltip>
              </Checkbox>
            </Row>
          )}
          {btn?.isNeed && (
            <Row>
              <label className="label-control">按钮标题</label>
              <Tooltip
                overlayClassName="tooltip-character-limit"
                placement="top"
                title={<span>{`最多${btn?.maxLength}个字符`}</span>}
                trigger="focus"
              >
                <Input
                  onChange={(e) =>
                    onChange(comKey, data.id, "btnName", e.target.value)
                  }
                  value={data.btnName}
                  maxLength={btn?.maxLength}
                />
              </Tooltip>
            </Row>
          )}
          {targetUrl?.isNeed && (
            <Row>
              <label className="label-control">跳转页面地址</label>
              <Input
                value={data.targetUrl}
                onChange={(e) =>
                  onChange(comKey, data.id, "targetUrl", e.target.value)
                }
                onBlur={(e) => {
                  !checkURL(e.target.value) && message.error("链接无效");
                }}
              />
            </Row>
          )}
        </div>
      )}
    </div>
  );
};
export default PropertyList;

const $$initialState = {
  pageConfig: {
    personalInfo: {},
    littleWallet: {},
    littleOrder: {},
    activeTask: {},
    entertainment: {},
    advertConfig: {},
    allTools: {},
  },
};
export default (state = $$initialState, action) => {
  const { comName, comType, comKey, value, index } = action.configData || {};
  let newState;
  switch (action.type) {
    case "PLATFORM_SET_PAGE_CONFIG":
      return { pageConfig: action.payload };
    case "CHANGE_VALUE":
      newState = _.cloneDeep(state);
      newState.pageConfig[comName][comType][comKey] = value;
      return newState;
    case "CHANGE_SECOND_VALUE":
      const { id, changeType } = action.configData;
      newState = _.cloneDeep(state);
      newState.pageConfig[comName][comType][comKey].forEach((item) => {
        if (item.id === id) {
          item[changeType] = value;
        }
      });
      return newState;
    case "CHANGE_ORDER_MODE":
      newState = _.cloneDeep(state);
      if (value) {
        const changeMode =
          comKey === "offlineMode_isShow"
            ? "onlineMode_isShow"
            : "offlineMode_isShow";
        newState.pageConfig[comName][comType][comKey] = true;
        newState.pageConfig[comName][comType][changeMode] = false;
      } else {
        newState.pageConfig[comName][comType][comKey] = false;
      }
      return newState;
    case "ADD_CUSTOM_DATA":
      const { addData } = action.configData;
      newState = _.cloneDeep(state);
      newState.pageConfig[comName][comType]["customList"].splice(
        index + 1,
        0,
        addData
      );
      return newState;
    case "DEL_CUSTOM_DATA":
      newState = _.cloneDeep(state);
      newState.pageConfig[comName][comType]["customList"].splice(index, 1);
      return newState;
    case "CHANGE_SHOW_ORDER_SHOW_STATE":
      const { code, bSelected } = action.configData;
      newState = _.cloneDeep(state);
      newState.pageConfig[comName][comType].forEach((item) => {
        if (item.code === code) {
          item["isShow"] = bSelected;
        }
      });
      return newState;
    case "CHANGE_DRAG_SORT":
      const { ele } = action.configData;
      newState = _.cloneDeep(state);
      let newShowOrder = [];
      ele.forEach((code) => {
        let ele = _.find(state.pageConfig[comName]["showOrder"], function(o) {
          return o.code == code;
        });
        ele && newShowOrder.push(ele);
      });
      newState.pageConfig[comName]["showOrder"] = newShowOrder;
      return newState;
    default:
      return state;
  }
};

原因：可能是由于轮播图数据是动态加载的，加载完成前后轮播图数量不一致

尝试解决：

  1. 在 state 中添加表示轮播图加载完成的数据

  2. 在轮播图数据加载完成时，修改该数据状态值为 true

  3. 只有在轮播图数据加载完成的情况下，才渲染轮播图组件

结果：无效

固定 slider-list 的高度
