# 店铺装修

# 介绍

老商城项目重构，后端服务用的老的

介绍：

  • 目前只给商城使用
  • U 商城有哪些客户：大企业一般买了 ERP，然后顺带买了营销云的其他产品
    • 富隆酒窖
    • 百兴盛酒业
    • ALBION 澳尔滨

关键特性：

  • 首次启用商城，需要初始化首页，进入店铺装修节点后，按提示确认完成初始化
  • 新建空白页面或者根据预置模板创建页面，进行装修
  • 拖放组件到编辑区，按组件类型添加图片、商品等数据，设置展示样式，预览最终效果
  • 页面保存后生效。也可以保存为草稿（不会影响实际页面），下次快速恢复编辑

# 1. 购物车

# 1.1 重新组合数据(handleGoodsData)

获取到的数据格式，数据虽然按照商家进行了分类，但是同一商家的商品放在了 shoppingCarts 。

具体问题及解决：

问题 解决
不易判断商品的类型 goodsType:1.普通商品 2.积分商品 3.周期购商品 4.有主商品的固定套餐和选配套餐（主+子）5.无主商品的固定套餐 6.搭配套餐
不易判断商品或者套餐是否失效 isValid 商品是否有效 true 有效，false 无效
不易找到同一个套餐的商品 goodsData:[ {},{} ] 无主商品的固定套餐和搭配套餐可能为多组,每组有 2 个以上商品
/**
 * 处理商品数据
 * goodsType:1.普通商品  2.积分商品 3.周期购商品  4.有主商品的固定套餐和选配套餐（主+子）
 *          5.无主商品的固定套餐  6.搭配套餐
 * 构建数据结构（多店铺）：
 * [
 *   {
 *     shopId:'',
 *     shopName:'',
 *     allProductGiftPreferDetail:[], // 赠品信息
 *     shopGoods:[
 *       {
 *         goodsType：1,
 *         sPackageUUID:'', // 区别商品
 *         isValid:true, // 商品是否有效 true 有效，false 无效
 *         goodsData:[   // 无主商品的固定套餐和搭配套餐可能为多组,每组有2个以上商品
 *         {},{}
 *         ],
 *       },
 *       {商品2},
 *       {商品3}
 *     ]
 *   },
 *   {店铺2}
 * ]
 * @param shoppingCartList
 */
handleGoodsData = (shoppingCartList) => {
  let handledGoods = [];
  shoppingCartList = _.filter(
    shoppingCartList,
    (v) => v.shoppingCarts && v.shoppingCarts.length > 0
  );
  shoppingCartList &&
    shoppingCartList.forEach((store) => {
      let storeItem = {
        shopId: store.shopId,
        shopName: store.shopName,
        allProductGiftPreferDetail: store.allProductGiftPreferDetail || [],
      };
      let shopGoods = [];
      store.shoppingCarts &&
        store.shoppingCarts.forEach((item) => {
          let newGoodsItem = {},
            goodsData = [];
          if (item.fixedSale === 0 || item.lPackageId === undefined) {
            if (item.productAttribute === 1) {
              if (item.iDispatchCycle) {
                //周期购商品
                newGoodsItem.goodsType = 3;
              } else {
                //普通商品
                newGoodsItem.goodsType = 1;
              }
            } else if (item.productAttribute === 2) {
              //积分商品
              newGoodsItem.goodsType = 2;
            }
            // 判断是否有效
            /**商品库存量为0，下架状态值为0则商品失效 */
            if (
              item.iQuantity <= 0 ||
              item.oSKU.iStatus == 0 ||
              item.oSKU?.lInventoryCount <= 0
            ) {
              newGoodsItem.isValid = false;
            } else {
              newGoodsItem.isValid = true;
            }
          } else if (item.fixedSale === 1 && item.packageDetail) {
            //有主商品的固定套餐和选配套餐（主+子）
            newGoodsItem.goodsType = 4;
            // 判断是否有效
            let isDetailValid = true;
            item.packageDetail.forEach((v) => {
              if (v.iProductNum <= 0) {
                isDetailValid = false;
              }
            });
            if (
              item.packageValid <= 0 ||
              item.oSKU.iStatus == 0 ||
              item.oSKU?.lInventoryCount <= 0 ||
              item.iQuantity <= 0 ||
              !isDetailValid
            ) {
              newGoodsItem.isValid = false;
            } else {
              newGoodsItem.isValid = true;
            }
          } else {
            //有相同sPackageUUID的归为一组
            let uuid = false;
            shopGoods.forEach((ele) => {
              if (ele.sPackageUUID === item.sPackageUUID) {
                uuid = true;
              }
            });
            if (uuid) {
              shopGoods.forEach((ele) => {
                if (ele.sPackageUUID === item.sPackageUUID) {
                  ele.goodsData.push(item);
                  // 判断是否有效
                  if (
                    item.packageValid == 0 ||
                    item.oSKU.iStatus == 0 ||
                    item.oSKU?.lInventoryCount <= 0 ||
                    item.iQuantity <= 0 ||
                    ele.isValid === false
                  ) {
                    ele.isValid = false;
                  } else {
                    ele.isValid = true;
                  }
                }
              });
              return;
            } else {
              if (
                item.fixedSale === 1 &&
                (item.bIsMain === undefined || item.bIsMain === false)
              ) {
                //无主商品的固定套餐  sPackageUUID归类
                newGoodsItem.goodsType = 5;
              } else {
                //搭配套餐
                newGoodsItem.goodsType = 6;
              }
              if (
                item.packageValid == 0 ||
                item.oSKU.iStatus == 0 ||
                item.oSKU?.lInventoryCount <= 0 ||
                item.iQuantity <= 0
              ) {
                newGoodsItem.isValid = false;
              } else {
                newGoodsItem.isValid = true;
              }
            }
          }
          // newGoodsItem.uuid = uuid()
          goodsData.push(item);
          newGoodsItem.goodsData = goodsData;
          newGoodsItem.sPackageUUID = item.sPackageUUID || item.id;
          shopGoods.push(newGoodsItem);
        });
      storeItem.shopGoods = shopGoods;
      handledGoods.push(storeItem);
    });
  return handledGoods;
};

# 1.3 互斥判断

其中，互斥判断情景：选中操作、全选、去结算

互斥判断规则：

  1. 开启商家模式的情况下，不同店铺商品不允许同时选中
  2. 不同配送方式不能同时选中
    1. 当开启平台配送或开启分销或社区团，进行判断
    2. 开启商家自行配送，不进行判断
  3. 积分商品不能与其他商品同时选中
  4. 周期购商品不能与其他商品同时选中
  5. 虚拟商品不能与普通商品同时选中

class SearchPage extends Component {
  handleHotKeyWordClick = (ele) => {
    if (!ele.targetUrl) {
      this.context.router.history.push({
        pathname: "/productList",
        state: { listType: 1, keyword: ele.name ?? "" },
      });
    } else {
      // 自定义
      if (ele.isSelfDefine) {
        if (/https?/.test(ele.targetUrl)) {
          location.href = mycb.rest.buildUrl(ele.targetUrl);
        } else {
          // channel/channelxxx
          let splitUrl = ele.targetUrl.split(/[/?=&]/);
          if (splitUrl.indexOf("channel") !== -1) {
            this.context.router.history.push(ele.targetUrl);
          }
          // /detail?goods_id=xxx
          else if (
            splitUrl.indexOf("detail") !== -1 &&
            splitUrl.indexOf("goods_id") !== -1
          ) {
            let goodsIdIndex = splitUrl.indexOf("goods_id");
            this.context.router.history.push({
              pathname: "/detail",
              state: { id: splitUrl[goodsIdIndex + 1] || "" },
            });
          }
          // /list?brand_id=xxx
          else if (splitUrl.indexOf("list") !== -1) {
            let listIndex = splitUrl.indexOf("list");
            if (splitUrl[listIndex + 1] && splitUrl[listIndex + 2]) {
              if (splitUrl[listIndex + 1] === "categoryid") {
                this.context.router.history.push({
                  pathname: "/productList",
                  state: { listType: 1, categoryid: splitUrl[listIndex + 2] },
                });
              } else {
                this.context.router.history.push({
                  pathname: "/productList",
                  state: {
                    listType: 1,
                    fieldType: splitUrl[listIndex + 1],
                    valuefrom: splitUrl[listIndex + 2],
                  },
                });
              }
            } else {
              this.context.router.history.push({
                pathname: "/productList",
                state: { listType: 1, keyword: ele.name ?? "" },
              });
            }
          } else {
            console.log("分情况跳转页面");
          }
        }
      } else {
        let splitUrl = ele.targetUrl.split(/[/?=]/);
        if (splitUrl[2] && splitUrl[3]) {
          if (splitUrl[2] === "categoryid") {
            this.context.router.history.push({
              pathname: "/productList",
              state: { listType: 1, categoryid: splitUrl[3] },
            });
          } else {
            this.context.router.history.push({
              pathname: "/productList",
              state: {
                listType: 1,
                fieldType: splitUrl[2],
                valuefrom: splitUrl[3],
              },
            });
          }
        } else {
          this.context.router.history.push({
            pathname: "/productList",
            state: { listType: 1, keyword: ele.name ?? "" },
          });
        }
      }
    }
  };
}

2）使用字符串内置方法 replace + 正则

String.prototype.replace() (opens new window)

  • 语法：str.replace(regexp|substr, newSubStr|function)
class SearchPage extends Component {
  constructor(props) {
    this.urlMapToPathname = {
      list: mycb.rest.buildInnerLinkUrl("/productList"),
      detail: mycb.rest.buildInnerLinkUrl("/detail"),
    };
  }
  handleUrl = (url, name) => {
    let routerParam;
    url.replace(/([^/]+)\?(\w+)=(\w+)/g, (_, type, key, value) => {
      if (type && key && value) {
        switch (type) {
          case "detail":
            routerParam = {
              pathname: this.urlMapToPathname.detail,
              state: { id: value },
            };
            break;
          case "list":
            if (key === "categoryid") {
              routerParam = {
                pathname: this.urlMapToPathname.list,
                state: { listType: 1, categoryid: value },
              };
            } else {
              routerParam = {
                pathname: this.urlMapToPathname.list,
                state: { listType: 1, fieldType: key, valuefrom: value },
              };
            }
            break;
          default:
            break;
        }
      } else {
        routerParam = {
          pathname: this.urlMapToPathname.list,
          state: { listType: 1, keyword: name ?? "" },
        };
      }
    });
    return routerParam;
  };
  // 点击热门搜索关键词
  handleHotKeyWordClick = (ele) => {
    if (!ele.targetUrl) {
      this.context.router.history.push({
        pathname: this.urlMapToPathname.list,
        state: { listType: 1, keyword: ele.name ?? "" },
      });
    } else {
      // 自定义
      if (ele.isSelfDefine) {
        if (/https?/.test(ele.targetUrl)) {
          location.href = mycb.rest.buildUrl(ele.targetUrl);
        } else if (/channel\//.test(ele.targetUrl)) {
          const routeUrl = mycb.rest.buildInnerLinkUrl(ele.targetUrl);
          this.context.router.history.push({ pathname: routeUrl });
        } else {
          this.context.router.history.push(
            this.handleUrl(ele.targetUrl, ele.name)
          );
        }
      } else {
        this.context.router.history.push(
          this.handleUrl(ele.targetUrl, ele.name)
        );
      }
    }
  };
}

// 点击语音图标
handleClickSpeech = () => {
  let self = this;
  wx.startRecord();
  self.setState({ isShowSpeechModal: true });
  wx.onVoiceRecordEnd({
    // 录音时间超过一分钟没有停止的时候会执行 complete 回调
    complete(res) {
      let localId = res.localId;
      self.recordKeySearch(localId);
    },
    fail(e) {
      Toast.fail(e.errMsg, 2);
      self.setState({ isShowSpeechModal: false });
    },
  });
};
// 关闭语音输入对话框
onCloseSpeechModal = () => {
  let self = this;
  wx.stopRecord({
    success(res) {
      let localId = res.localId;
      self.recordKeySearch(localId);
    },
    fail(e) {
      Toast.fail(e.errMsg, 2);
      self.setState({ isShowSpeechModal: false });
    },
  });
};
recordKeySearch = (vid) => {
  let self = this;
  wx.translateVoice({
    localId: vid,
    isShowProgressTips: 1,
    success(res) {
      let tsResult = res.translateResult;
      if (tsResult.indexOf("。") >= 0) {
        tsResult = tsResult.replace("。", "");
      }
      self.setState({ inputText: tsResult, isShowSpeechModal: false });
    },
    fail(e) {
      Toast.fail(e.errMsg, 2);
      self.setState({ isShowSpeechModal: false });
    },
  });
};
// 当关闭对话框，结束录音
onClickModalClose = () => {
  this.setState({ isShowSpeechModal: false });
  wx.stopRecord();
};

# （4）问题及解决

遇到的问题：

  • 安卓手机，二级路由 不能调用 wx.startRecord() 等相关 API

解决：

  • 在当前页面 调 client/Pay/getWeChatConfig 接口
    • 初始化微信？

步骤三：通过 config 接口注入权限验证配置 所有需要使用 JS-SDK 的页面必须先注入配置信息，否则将无法调用（同一个 url 仅需调用一次，对于变化 url 的 SPA 的 web app 可在每次 url 变化时进行调用,目前 Android 微信客户端不支持 pushState 的 H5 新特性，所以使用 pushState 来实现 web app 的页面会导致签名失败，此问题会在 Android6.2 中修复）。

export default class SearchPage extends Component {
  componentDidMount() {
    let isAndroid = /android/i.test(navigator.userAgent);
    if (mycb.utils.isWeChat() && isAndroid) {
      this.getWeChatConfig();
    }
  }
  getWeChatConfig = async () => {
    const config = {
      url: "client/Pay/getWeChatConfig",
      method: "POST",
      params: { url: location.href },
      options: { uniform: true },
    };
    const result = await proxy(config);
    if (result.code !== 200) {
      return;
    } else {
      const { appId, timestamp, nonceStr, signature } = result.data;
      // 步骤三：通过config接口注入权限验证配置
      wx.config({
        debug: false,
